        get_ci <- function(n, p) {
    f1 <- qf(0.025, 2 * n * p, 2 * (n - n * p + 1))
    f2 <- qf(0.975, 2 * (n * p + 1), 2 * n * (1 - p))
    low <- (n * p * f1) / (n * p * f1 + n * (1 - p) + 1)
    up <- ((n * p + 1) * f2) / ((n * p + 1) * f2 + n * (1 - p))
    return(c(low, up))
}
# input n and p to get C-P confidence interval
n <- c(10, 10, 100, 100, 100, 1000, 1000)
p <- c(0.5, 0.9, 0.5, 0.9, 0.99, 0.99, 0.999)
for (i in 1:7) {
    print(get_ci(n[i], p[i]))
}
