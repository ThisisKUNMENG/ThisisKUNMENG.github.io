# In the analysis, we treat median house value "medv" as the response and the
# other thirteen variables as the covariates. Note that "chas" is a categorical
# variable, so we ignore it in the following analysis.
#
# First, we load the data set and drop the "chas" variable.
#
# ------------------------------------------------------------------------------

install.packages(setdiff("mlbench", rownames(installed.packages())))
install.packages(setdiff("MASS", rownames(installed.packages())))
install.packages(setdiff("car", rownames(installed.packages())))
# install "mlbench", "MASS", "car" package if they are not installed

library(mlbench) # load "mlbench" package
library(MASS) # load "MASS" package
library(car) # load "car" package

data("BostonHousing") # load the "BostonHousing" data set
? BostonHousing # check the basic information of the "BonstonHousing" data set

boston_data <- subset(BostonHousing, select = -c(chas))
# Rename the dataset and remove the "chas" variable

# ------------------------------------------------------------------------------
#
# Next, we conduct some descriptive analysis.
#
# ------------------------------------------------------------------------------

summary(boston_data) # output the summary statistics of the dataset

# ------------------------------------------------------------------------------
#
# Next, we conduct multivariate linear regression analysis.
#
# ------------------------------------------------------------------------------

lm_boston <- lm(medv ~ ., data = boston_data) # fit a linear regression model
summary(lm_boston) # output the fitting results

e <- resid(lm_boston) # calculate the residuals
sigma2 <- sum(e^2) / (nrow(boston_data) - ncol(boston_data) - 1)
# estimate the random error variance: sigma2

x_mat <- as.matrix(boston_data[, 1:12]) # extract design matrix from dataset
x_mat <- cbind(1, x_mat) # add a column of ones to the right of design matrix
x_t_x <- t(x_mat) %*% x_mat # calculate X'X
cov_beta <- sigma2 * solve(x_t_x) # calculate the covariance matrix of \hat{β}
cov_beta # output the covariance matrix of \hat{β}

anova(lm_boston)
# carry out the F-test for each covariate and output the results

lm_boston_drop <- lm(medv ~ . - nox, data = boston_data)
# drop "nox" since it is the most insignificant covariate in the anova results
summary(lm_boston_drop) # output the fitting results after dropping "nox"

# ------------------------------------------------------------------------------
#
# Next, we conduct diagnostics on the original model "lm_boston".
# Heteroscedasticity is found and Boxcox transformation is used to deal with it.
#
# ------------------------------------------------------------------------------

e_abs <- abs(e) # calculate the absolute values of the residuals
rho_hat <- c()
rho_p <- c() # initialize vectors; store correlation coefficients and p-values
boston_data_covariate <- boston_data[, -13] # extract covariates from dataset
for (i in seq_len(ncol(boston_data_covariate))) {
    temp <- cor.test(boston_data_covariate[, i], e_abs, method = "spearman")
    rho_hat <- c(rho_hat, temp$estimate)
    rho_p <- c(rho_p, temp$p.value)
}
# calculate the correlation coefficient between the residuals and each covariate

data.frame(cbind(rho_hat, rho_p), row.names = colnames(boston_data_covariate))
# output the correlation coefficients and p-values

bc <- boxcox(medv ~ ., data = boston_data, lambda = seq(-2, 2, 0.01))
# Since "medv" is all positive, we can use Boxcox transformation to deal with
# the heteroscedasticity problem shown above.
# The lambda value range is [-2, 2] and the step size is 0.01.
lambda <- bc$x[which.max(bc$y)]
# choose the lambda value that maximizes the likelihood function
lambda # output the lambda value
y_bc <- (boston_data$medv^lambda - 1) / lambda # calculate transformed y value

lm_boston_bc <- lm(y_bc ~ . - medv, data = boston_data)
# fit a linear model using transformed y values
summary(lm_boston_bc) # output the fitting results

e_bc_abs <- abs(resid(lm_boston_bc)) # same process to compute coef and p-value
rho_bc_hat <- c()
rho_bc_p <- c()
for (i in seq_len(ncol(boston_data_covariate))) {
    temp <- cor.test(boston_data_covariate[, i], e_bc_abs, method = "spearman")
    rho_bc_hat <- c(rho_bc_hat, temp$estimate)
    rho_bc_p <- c(rho_bc_p, temp$p.value)
}

data.frame(cbind(rho_bc_hat, rho_bc_p),
           row.names = colnames(boston_data_covariate)) 
# output the correlation coefficients and p-values

# ------------------------------------------------------------------------------
#
# Next, we detect multicollinearity and use ridge regression to deal with it.
#
# ------------------------------------------------------------------------------

mean(vif(lm_boston)) # calculate the average of VIFs
# It is much greater than 1, indicating the presence of multicollinearity.

cor_mat <- cor(boston_data_covariate) # calculate the correlation matrix
kappa(cor_mat, exact = TRUE) # calculate the condition number
# It is extremely large, indicating the existence of multicollinearity.

boston_data_std <- data.frame(scale(boston_data)) # standardize data
ridge_boston <- lm.ridge(medv ~ ., data = boston_data_std,
                         lambda = seq(0, 3, 0.1)) # perform ridge regression
beta <- coef(ridge_boston) # assign parameter estimates of ridge regression
beta # output the parameter estimates

k <- ridge_boston$lambda # assign candidate tuning parameters to k

plot(k, k,
     type = "n", xlab = "Tuning Parameter k", ylab = "Coefficients",
     ylim = c(-0.5, 0.5)) # create a graphics area without points and lines

linetype <- 1 : 5 # specify five line types used in the plot
char <- 18 : 22 # specify five point shapes used in the plot
for (i in seq_len(ncol(beta)))
  lines(k, beta[, i], type = "o", lty = linetype[i %% 5], pch = char[i %% 5], 
        cex = 0.75, col = i) # add ridge traces to the blank plot

legend("topright", legend = colnames(beta), pch = char, lty = linetype,
       cex = 0.8, col = seq_len(ncol(beta)), ncol = 5) # add legends
