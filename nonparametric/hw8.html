    from scipy.stats import norm

def risk_t(lam, theta_i, sigma=1):
    p_i3p_i1minus = 1 - norm.cdf((lam-theta_i)/sigma) - norm.cdf((-lam-theta_i)/sigma)
    p_i3p_i1plus = 1 - norm.cdf((lam-theta_i)/sigma) + norm.cdf((-lam-theta_i)/sigma)
    part1 = 4 * sigma**2 * np.sum(p_i3p_i1minus * theta_i)
    part2 = -2 * lam * sigma**2 * np.sum(p_i3p_i1plus)
    part3 = np.sum(theta_i**2)
    part4 = np.sum(
        p_i3p_i1plus *
        (theta_i**4 + 6 * theta_i**2 * sigma**2 + 3 * sigma**4 +
         (6 * lam**2 + 4 * lam) * (sigma**2 + theta_i**2) + lam**4)
    )
    part5 = -np.sum(
        p_i3p_i1minus *
        (2 * (2*lam+1) * (theta_i**3 + 3 * theta_i * sigma**2) +
         2 * lam**2 * (2*lam+1) * theta_i)
    )
    return part1 + part2 + part3 + part4 + part5

    from estimate import Estimate
from tqdm import tqdm
class NormalMean(Estimate):
    def __init__(self, y: pd.Series, x: pd.Series):
        super().__init__(y, x)
        self.data = None
        self.method = 'Normal Mean'
        self.process_same_value()
        self.x_old = self.x
        self._scale_x()
        self.z = self.get_z(self.x.values)
        self.b = (1 - self.n / np.sum(self.z**2)) * np.ones(self.n)
        self.jn = int(np.ceil(self.n/4))
        self.j_hat = None
        self.var = self.variance_estimate()
        self.optimal_j_hat()

    def process_same_value(self):
        self.data = pd.DataFrame({"x": self.x, "y": self.y})
        self.x = pd.Series(self.data["x"].value_counts().index.sort_values().values)
        self.y = pd.Series([self.data[self.data["x"] == i]["y"].mean() for i in self.x])
        self.n = len(self.x)

    def _scale_x(self):
        self.x = (self.x - self.x.min()) / (self.x.max() - self.x.min())

    @staticmethod
    def cosine_basis(i: int):
        def f(xxx: int | float | np.ndarray) -> np.ndarray:
            if i == 1:
                return np.ones_like(xxx)
            return np.array([np.sqrt(2) * np.cos((i-1) * np.pi * xxx)])
        return f

    def get_z(self, x: np.ndarray) -> np.ndarray:
        z = np.zeros(self.n)
        for j in range(self.n):
            z[j] = 1/self.n * np.sum(self.cosine_basis(j+1)(x) * self.y.values)
        return z

    def estimate(self, x) -> float | np.ndarray:
        if isinstance(x, np.ndarray):
            return np.array([self.estimate(x[i]) for i in range(len(x))])
        s = 0
        for i in range(self.j_hat):
            s += self.z[i] * self.cosine_basis(i+1)(x)
        return s

    def variance_estimate(self) -> float:
        return 1/(self.n - self.jn) * np.sum([self.z[i]**2 for i in range(self.n-self.jn, self.n)])

    def risk(self, j: int):
        part1 = j * self.var / self.n
        part2 = np.sum([np.max([(self.z[j]**2 - self.var/self.n), 0]) for j in range(j, self.n)])
        return part1 + part2

    def optimal_j_hat(self):
        j_hat = np.argmin([self.risk(j) for j in range(1, self.n)])
        self.j_hat = j_hat
        return j_hat

    def l_x(self, x) -> np.ndarray:
        li_x = np.zeros(self.n)
        for i in range(self.n):
            li_x[i] = 1/self.n * np.sum(
                np.array(
                    [float(self.b[j] * self.cosine_basis(j+1)(x) * self.cosine_basis(j+1)(self.x.values[i]))
                     for j in range(self.n)]
                )
            )
        return li_x

    def confidence_band(self):
        self.c = 1.96
        sigma = np.sqrt(self.variance_estimate())

        def band(x):
            s = self.c * np.sqrt(sigma) * np.linalg.norm(self.l_x(x))
            return [float(self.estimate(x) - s),
                    float(self.estimate(x) + s),
                    float(self.estimate(x))]

        return band

    def draw_confidence_band(self, flag=True, ax=None):
        if ax is None:
            ax = plt
        band = self.confidence_band()
        x = np.linspace(self.x_old.min(), self.x_old.max(), 100)
        new_x = (x - self.x_old.min()) / (self.x_old.max() - self.x_old.min())
        y = np.array([band(i) for i in tqdm(new_x, desc=self.method + " confidence band")])
        ax.plot(self.x_old, self.y, "o", color="grey", ms=2)
        ax.fill_between(x, y[:, 0], y[:, 1], alpha=.5, color='red')
        ax.plot(x, y[:, 2], color="red", label="estimate")
        ax.legend()
        return [x, y[:, 0], y[:, 1]]
